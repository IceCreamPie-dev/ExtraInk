1. 간단한 대화 스크립트 작성

using UnityEngine;
using DeulPie.ExtraInk.Core.Managers;
public class SimpleDialogueExample : MonoBehaviour
{
private void Start()
{
StartDialogue();
}
private void StartDialogue()
{
var dialogueManager = DialogueManager.Instance;
// 이벤트 구독
dialogueManager.OnChoiceSelected += OnChoiceSelected;
// 첫 번째 대화
dialogueManager.ShowDialogue("안녕하세요! ExtraInk에 오신 것을 환영합니다.", "시스템");
// 2초 후 선택지 표시
Invoke(nameof(ShowChoices), 2f);
}
private void ShowChoices()
{
var dialogueManager = DialogueManager.Instance;
string[] choices = {
"ExtraInk에 대해 더 알아보기",
"바로 시작하기",
"나중에 하기"
};
dialogueManager.ShowChoices(choices);
}
private void OnChoiceSelected(int index, string choice)
{
var dialogueManager = DialogueManager.Instance;
switch (index)
{
case 0:
dialogueManager.ShowDialogue(
"ExtraInk는 Unity를 위한 고성능 비주얼 노벨 프레임워크입니다!",
"시스템");
break;
case 1:
dialogueManager.ShowDialogue("좋습니다! 게임을 시작하겠습니다.", "시스템");
break;
case 2:
dialogueManager.ShowDialogue("언제든 다시 오세요!", "시스템");
break;
}
}
}

2. Ink 스토리와 연동

using Ink.Runtime;
using DeulPie.ExtraInk.Core.Managers;
public class InkDialogueExample : MonoBehaviour
{
[SerializeField] private TextAsset inkStoryAsset;
private Story inkStory;
private void Start()
{
// Ink 스토리 초기화
inkStory = new Story(inkStoryAsset.text);
// ExtraInk와 연동
StartInkStory();
}
private void StartInkStory()
{
var dialogueManager = DialogueManager.Instance;
while (inkStory.canContinue)
{
string text = inkStory.Continue();
// 화자 정보 추출 (태그 사용)
string speaker = ExtractSpeaker(inkStory.currentTags);
// ExtraInk로 대화 표시
dialogueManager.ShowDialogue(text, speaker);
// 선택지 처리
if (inkStory.currentChoices.Count > 0)
{
ShowInkChoices();
break;
}
}
}
private void ShowInkChoices()
{
var dialogueManager = DialogueManager.Instance;
var choices = new string[inkStory.currentChoices.Count];
for (int i = 0; i < choices.Length; i++)
{
choices[i] = inkStory.currentChoices[i].text;
}
dialogueManager.ShowChoices(choices);
dialogueManager.OnChoiceSelected += OnInkChoiceSelected;
}
private void OnInkChoiceSelected(int index, string choice)
{
var dialogueManager = DialogueManager.Instance;
dialogueManager.OnChoiceSelected -= OnInkChoiceSelected;
// Ink에 선택 전달
inkStory.ChooseChoiceIndex(index);
// 스토리 계속
StartInkStory();
}
private string ExtractSpeaker(List<string> tags)
{
foreach (string tag in tags)
{
if (tag.StartsWith("speaker:"))
{
return tag.Substring(8); // "speaker:" 제거
}
}
return "내레이터";
}
}

🎭 캐릭터 추가하기

1. 캐릭터 프리팹 생성

// 캐릭터 컴포넌트 스크립트
using UnityEngine;
using DeulPie.ExtraInk.Core.Performance;
public class SimpleCharacter : PerformanceOptimizedCharacter
{
[Header("캐릭터 설정")]
[SerializeField] private Sprite[] emotionSprites;
[SerializeField] private SpriteRenderer spriteRenderer;
private void Awake()
{
base.Awake();
spriteRenderer = GetComponent<SpriteRenderer>();
}
public void SetEmotion(string emotion)
{
// 감정에 따른 스프라이트 변경
switch (emotion.ToLower())
{
case "happy":
spriteRenderer.sprite = emotionSprites[0];
break;
case "sad":
spriteRenderer.sprite = emotionSprites[1];
break;
case "angry":
spriteRenderer.sprite = emotionSprites[2];
break;
default:
spriteRenderer.sprite = emotionSprites[0];
break;
}
}
}

2. 비동기 오디오 로딩

using System.Threading;
using System.Threading.Tasks;
using UnityEngine;
using DeulPie.ExtraInk.Core.Performance;
public class AsyncAudioExample : MonoBehaviour
{
private async void Start()
{
await LoadAndPlayAudioAsync();
}
private async Task LoadAndPlayAudioAsync()
{
var performanceManager = PerformanceManager.Instance;
// 비동기로 오디오 클립 로드
var audioClip = await performanceManager.LoadAudioClipAsync(
"Audio/BGM/main_theme",
LoadAudioClipAsync
);
if (audioClip != null)
{
var audioManager = AudioManager.Instance;
audioManager.PlayBGM("main_theme");
}
}
private async Task<AudioClip> LoadAudioClipAsync(string path, CancellationToken token)
{
// Resources에서 비동기 로드
var request = Resources.LoadAsync<AudioClip>(path);
while (!request.isDone && !token.IsCancellationRequested)
{
await Task.Yield();
}
return request.asset as AudioClip;
}
}

🚀 성능 최적화 활성화

1. 메모리 풀링 설정

using UnityEngine;
using DeulPie.ExtraInk.Core.Performance;
using DeulPie.ExtraInk.Core.Performance.MemoryPooling;
public class PerformanceSetup : MonoBehaviour
{
[Header("풀링 설정")]
[SerializeField] private GameObject effectPrefab;
[SerializeField] private GameObject uiElementPrefab;
private void Start()
{
SetupPooling();
SetupAsyncProcessing();
MonitorPerformance();
}
private void SetupPooling()
{
var performanceManager = PerformanceManager.Instance;
// 이펙트 풀 생성
var effectPool = performanceManager.CreateEffectPool(effectPrefab, 20);
// UI 요소 풀 생성
var uiPool = performanceManager.CreateUIElementPool(
uiElementPrefab.GetComponent<RectTransform>(), 10);
Debug.Log("메모리 풀링이 설정되었습니다.");
}
private void SetupAsyncProcessing()
{
var performanceManager = PerformanceManager.Instance;
// 프레임 예산 설정 (60fps 기준)
performanceManager.SetFrameBudget(16.67f);
Debug.Log("비동기 처리가 설정되었습니다.");
}
private void MonitorPerformance()
{
var performanceManager = PerformanceManager.Instance;
// 성능 메트릭 모니터링
performanceManager.OnPerformanceMetricsUpdated += metrics => {
if (metrics.fps < 30f)
{
Debug.LogWarning($"낮은 FPS 감지: {metrics.fps:F1}");
performanceManager.OptimizePerformance();
}
};
}
}

2. 배치 처리 예제

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using DeulPie.ExtraInk.Core.Performance;
public class BatchProcessingExample : MonoBehaviour
{
private void Start()
{
StartCoroutine(ProcessLargeDataSet());
}
private IEnumerator ProcessLargeDataSet()
{
// 대량 데이터 생성
var largeDataSet = new List<int>();
for (int i = 0; i < 10000; i++)
{
largeDataSet.Add(i);
}
Debug.Log("대량 데이터 배치 처리 시작...");
// 배치 처리
yield return PerformanceManager.Instance.ProcessDialogueDataBatch(
largeDataSet,
processAction: data => {
// 각 데이터 처리
int result = (int)data * 2;
},
onProgress: progress => {
Debug.Log($"진행률: {progress * 100:F1}%");
}
);
Debug.Log("배치 처리 완료!");
}
}

🎯 통합 예제

모든 기능을 통합한 완전한 예제입니다.

using System.Collections;
using UnityEngine;
using DeulPie.ExtraInk.Core.Managers;
using DeulPie.ExtraInk.Core.Performance;
public class CompleteExample : MonoBehaviour
{
[Header("프리팹들")]
[SerializeField] private GameObject characterPrefab;
[SerializeField] private AudioClip backgroundMusic;
private void Start()
{
StartCoroutine(RunCompleteDemo());
}
private IEnumerator RunCompleteDemo()
{
// 1. 시스템 초기화
yield return InitializeSystem();
// 2. BGM 시작
var audioManager = AudioManager.Instance;
audioManager.PlayBGM("demo_bgm", 0.6f);
// 3. 캐릭터 등장
var characterManager = CharacterManager.Instance;
characterManager.ShowCharacter("demo_character", "happy", "center");
yield return new WaitForSeconds(1f);
// 4. 대화 시작
var dialogueManager = DialogueManager.Instance;
dialogueManager.ShowDialogue(
"안녕하세요! ExtraInk v2.0의 완전한 데모입니다.",
"데모 캐릭터");
yield return new WaitForSeconds(3f);
// 5. 선택지 표시
string[] choices = {
"성능 테스트 보기",
"캐릭터 애니메이션 보기",
"오디오 테스트 보기"
};
dialogueManager.ShowChoices(choices);
dialogueManager.OnChoiceSelected += OnDemoChoiceSelected;
Debug.Log("완전한 데모가 시작되었습니다!");
}
private IEnumerator InitializeSystem()
{
var performanceManager = PerformanceManager.Instance;
// 캐릭터 풀 생성
var characterPool = performanceManager.CreateCharacterPool(
characterPrefab.GetComponent<PerformanceOptimizedCharacter>(), 5);
// 성능 모니터링 시작
performanceManager.OnPerformanceMetricsUpdated += metrics => {
Debug.Log($"FPS: {metrics.fps:F1}, Memory: {metrics.memoryUsage / 1024 / 1024}MB");
};
yield return new WaitForSeconds(0.5f);
Debug.Log("시스템 초기화 완료");
}
private void OnDemoChoiceSelected(int index, string choice)
{
var dialogueManager = DialogueManager.Instance;
dialogueManager.OnChoiceSelected -= OnDemoChoiceSelected;
switch (index)
{
case 0:
StartCoroutine(ShowPerformanceDemo());
break;
case 1:
StartCoroutine(ShowCharacterDemo());
break;
case 2:
StartCoroutine(ShowAudioDemo());
break;
}
}
private IEnumerator ShowPerformanceDemo()
{
var dialogueManager = DialogueManager.Instance;
dialogueManager.ShowDialogue("성능 시스템을 테스트합니다...", "시스템");
// 성능 최적화 실행
PerformanceManager.Instance.OptimizePerformance();
yield return new WaitForSeconds(2f);
var metrics = PerformanceManager.Instance.GetCurrentMetrics();
dialogueManager.ShowDialogue(
$"현재 FPS: {metrics.fps:F1}, 메모리: {metrics.memoryUsage / 1024 / 1024}MB",
"시스템");
}
private IEnumerator ShowCharacterDemo()
{
var characterManager = CharacterManager.Instance;
var dialogueManager = DialogueManager.Instance;
dialogueManager.ShowDialogue("캐릭터 애니메이션을 보여드리겠습니다!", "데모 캐릭터");
yield return new WaitForSeconds(1f);
// 감정 변화
characterManager.ChangeEmotion("demo_character", "surprised");
yield return new WaitForSeconds(1f);
// 위치 이동
characterManager.MoveCharacter("demo_character", "left", 1.5f);
yield return new WaitForSeconds(2f);
dialogueManager.ShowDialogue("어떠세요? 멋지죠?", "데모 캐릭터");
}
private IEnumerator ShowAudioDemo()
{
var audioManager = AudioManager.Instance;
var dialogueManager = DialogueManager.Instance;
dialogueManager.ShowDialogue("오디오 시스템을 테스트해보겠습니다.", "시스템");
yield return new WaitForSeconds(1f);
// 효과음 재생
audioManager.PlaySFX("demo_sfx");
yield return new WaitForSeconds(1f);
// 볼륨 조절
audioManager.SetMasterVolume(0.3f);
yield return new WaitForSeconds(1f);
audioManager.SetMasterVolume(0.8f);
dialogueManager.ShowDialogue("오디오 테스트 완료!", "시스템");
}
}
